// ============================================================ // WARBA INSURANCE - ATTENDANCE SYSTEM // Google Apps Script Backend // Developed by: Syed Usama Najeeb — Digital Transformation Specialist // ============================================================ const SPREADSHEET_ID = '1vJ4q4hwAJRnUDp5aDb2yH0DQlqvWZEKAjzeNVCFdkKk'; const STUDENTS_SHEET = 'Students'; const ATTENDANCE_SHEET = 'Attendance'; const FEEDBACK_SHEET = 'Feedback'; // ============================================================ // HANDLE POST REQUEST // ============================================================ function doPost(e) { try { const data = JSON.parse(e.postData.contents); const mode = data.mode; const mobile = data.mobile.trim(); const feedback = data.feedback || ''; const timestamp = new Date(); const student = findStudent(mobile); if (!student) { return respond(false, null, 'Mobile number not found. Please check your number or contact your administrator.'); } if (mode === 'checkin') { return handleCheckIn(student, mobile, timestamp); } else if (mode === 'checkout') { return handleCheckOut(student, mobile, feedback, timestamp); } else { return respond(false, null, 'Invalid mode.'); } } catch (err) { return respond(false, null, 'An error occurred: ' + err.message); } } // ============================================================ // HANDLE GET REQUEST — required for CORS preflight // ============================================================ function doGet(e) { const mode = e.parameter.mode || ''; const mobile = e.parameter.mobile || ''; const feedback = e.parameter.feedback || ''; const timestamp = new Date(); if (!mobile || !mode) { return respond(false, null, 'Missing parameters.'); } const student = findStudent(mobile); if (!student) { return respond(false, null, 'Mobile number not found. Please check your number or contact your administrator.'); } if (mode === 'checkin') { return handleCheckIn(student, mobile, timestamp); } else if (mode === 'checkout') { return handleCheckOut(student, mobile, feedback, timestamp); } return respond(false, null, 'Invalid mode.'); } // ============================================================ // FIND STUDENT BY MOBILE NUMBER // ============================================================ function findStudent(mobile) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(STUDENTS_SHEET); const data = sheet.getDataRange().getValues(); const cleanMobile = mobile.replace(/[\s\-\(\)]/g, ''); for (let i = 1; i < data.length; i++) { const storedMobile = String(data[i][1]).replace(/[\s\-\(\)]/g, ''); if (storedMobile === cleanMobile) { return { name: data[i][0], mobile: data[i][1], email: data[i][2] || '', batch: data[i][3] || '', row: i + 1 }; } } return null; } // ============================================================ // HANDLE CHECK-IN // ============================================================ function handleCheckIn(student, mobile, timestamp) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(ATTENDANCE_SHEET); const today = Utilities.formatDate(timestamp, Session.getScriptTimeZone(), 'yyyy-MM-dd'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { const rowDate = data[i][3] ? String(data[i][3]) : ''; const rowMobile = String(data[i][2]).replace(/[\s\-\(\)]/g, ''); const cleanMobile = mobile.replace(/[\s\-\(\)]/g, ''); if (rowDate === today && rowMobile === cleanMobile && data[i][4]) { return respond(true, student.name, 'You have already checked in today at ' + String(data[i][4]) + '. See you later!'); } } const lastRow = Math.max(sheet.getLastRow(), 1); const dateStr = Utilities.formatDate(timestamp, Session.getScriptTimeZone(), 'yyyy-MM-dd'); const timeStr = Utilities.formatDate(timestamp, Session.getScriptTimeZone(), 'HH:mm:ss'); sheet.appendRow([lastRow, student.name, student.mobile, dateStr, timeStr, 'Checked In', '']); if (sheet.getRange(1, 1).getValue() === '') { sheet.getRange(1, 1).setValue('Record ID'); sheet.getRange(1, 2).setValue('Student Name'); sheet.getRange(1, 3).setValue('Mobile Number'); sheet.getRange(1, 4).setValue('Check-in Date'); sheet.getRange(1, 5).setValue('Check-in Time'); sheet.getRange(1, 6).setValue('Status'); sheet.getRange(1, 7).setValue('Check-out Time'); } return respond(true, student.name, 'Welcome, ' + student.name + '! Your check-in has been recorded at ' + timeStr + '.'); } // ============================================================ // HANDLE CHECK-OUT // ============================================================ function handleCheckOut(student, mobile, feedback, timestamp) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const attendanceSheet = ss.getSheetByName(ATTENDANCE_SHEET); const feedbackSheet = ss.getSheetByName(FEEDBACK_SHEET); const today = Utilities.formatDate(timestamp, Session.getScriptTimeZone(), 'yyyy-MM-dd'); const data = attendanceSheet.getDataRange().getValues(); const timeStr = Utilities.formatDate(timestamp, Session.getScriptTimeZone(), 'HH:mm:ss'); const dateStr = Utilities.formatDate(timestamp, Session.getScriptTimeZone(), 'yyyy-MM-dd'); let checkInRowIndex = -1; for (let i = 1; i < data.length; i++) { const rowDate = String(data[i][3]); const rowMobile = String(data[i][2]).replace(/[\s\-\(\)]/g, ''); const cleanMobile = mobile.replace(/[\s\-\(\)]/g, ''); if (rowDate === today && rowMobile === cleanMobile) { checkInRowIndex = i + 1; break; } } if (checkInRowIndex > -1) { attendanceSheet.getRange(checkInRowIndex, 6).setValue('Checked Out'); attendanceSheet.getRange(checkInRowIndex, 7).setValue(timeStr); } if (feedback) { const feedbackLastRow = Math.max(feedbackSheet.getLastRow(), 1); feedbackSheet.appendRow([feedbackLastRow, student.name, student.mobile, dateStr, feedback, '', timeStr]); } const message = checkInRowIndex > -1 ? 'Goodbye, ' + student.name + '! Your check-out has been recorded at ' + timeStr + '. Thank you for today!' : 'Goodbye, ' + student.name + '! Your checkout has been saved.'; return respond(true, student.name, message); } // ============================================================ // RESPONSE HELPER // ============================================================ function respond(success, name, message) { const output = { success: success, name: name, message: message }; return ContentService .createTextOutput(JSON.stringify(output)) .setMimeType(ContentService.MimeType.JSON); }
